    private zeichneNurFavoriten(stations: Station[], map: L.Map): void {
        // Icon für die Favoriten
        var favorit = L.icon({ iconUrl: 'https://cdn-icons-png.flaticon.com/512/121/121724.png',iconSize: [25, 30], popupAnchor: [0, -13]});
